1function [feas, xOpt, uOpt, JOpt, xD] = solve_cftoc(N, x0, xN, t, xL, xU, uL, uU)
 2
 3x = sdpvar(3, N+1);
 4u = sdpvar(2, N);
 5dt = 1;
 6Q = [1 0; 0 1];
 7P = 5*Q;
 8R = 0.1;
 9
10xd = get_desired(t+N*dt);
11Cost = (x(1:2, N+1)-xd)' * P * (x(1:2, N+1)-xd);
12
13if ~isempty(xN)
14    Constr = [x(:, N+1) == xN];
15else
16    Constr = [];
17end
18
19Constr = Constr + [x(:, 1) == x0];
20xD = [];
21for i = 1:N
22    xd = get_desired(t+(i-1)*dt);
23    xD = [xD, xd];
24    Cost = Cost + (x(1:2, i)-xd)'*Q*(x(1:2, i)-xd) + u(:, i)'*R*u(:, i);
25    Constr = Constr + [x(:, i+1) == [x(1,i) + u(1,i)*cos(x(3,i)+u(2,i)*dt)*dt; 
26                                     x(2,i) + u(1,i)*sin(x(3,i)+u(2,i)*dt)*dt;
27                                     x(3,i) + u(2,i)*dt]];
28    Constr = Constr + [uL<=u(:, i)<=uU];
29    if i <= N-1
30        Constr = Constr + [-0.1<=u(1,i+1)-u(1,i)<=0.1];
31        Constr = Constr + [-0.1<=u(2,i+1)-u(2,i)<=0.1];
32    end
33end
34
35xD = [xD, get_desired(t+N*dt)];
36options = sdpsettings('verbose', 0);
37sol = optimize(Constr,Cost,options);
38
39feas = sol.problem == 0;
40if feas
41    xOpt = double(x);
42    uOpt = double(u);
43    JOpt = double(Cost);
44else
45    xOpt = [];
46    uOpt = [];
47    JOpt = [];
48end
49    
50end
